//сумма чётных элементов ряда Фибоначчи меньших 4 млн
#include <iostream>
 
using namespace std;
 
int main() {
    int a=1;
    int b=2;
    int c=0;
    unsigned long sum=2;
    while (c<=4000000) {
       c=a+b;
       if (c%2==0) sum+=c;
       a=b;
       b=c;
   }

    cout<<sum<<endl;
 
return 0;
}
