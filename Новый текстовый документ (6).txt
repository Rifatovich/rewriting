//Инициалы из ФИО
#include <iostream>

using namespace std;

int main() 
{
    string str;
    getline(cin, str);
    int i;
    for (i = 0; str[i] != ' '; i++)
    {
        cout << str[i];
    }
    cout << ' ' << str[i+1] << ". ";
    for (i++; str[i] != ' '; i++);
    cout << str[i + 1] << '.';
    return 0;
}
